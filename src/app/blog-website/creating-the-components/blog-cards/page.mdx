## Introduction

Blog cards are a crucial component of any blog website, providing a visually appealing way to display blog posts. They typically include the title, a brief description, and an image or thumbnail.

## Creating the Blog Card Component

To create a blog card component, we will use React and Tailwind CSS. The component will accept props for the title, description, image URL, and a link to the full post.

Since we are using Next.js with MDX, we can create a new component file in the `components` directory.

```jsx
// components/BlogCard.js
import React from "react";
import Link from "next/link";
const BlogCard = ({ title, description, imageUrl, link }) => {
  return (
    <div className="max-w-sm rounded overflow-hidden shadow-lg">
      <img className="w-full" src={imageUrl} alt={title} />
      <div className="px-6 py-4">
        <div className="font-bold text-xl mb-2">{title}</div>
        <p className="text-gray-700 text-base">{description}</p>
      </div>
      <div className="px-6 pt-4 pb-2">
        <Link href={link}>
          <a className="text-blue-500 hover:text-blue-800">Read More</a>
        </Link>
      </div>
    </div>
  );
};
export default BlogCard;
```

This component uses Tailwind CSS for styling and accepts props for the title, description, image URL, and link to the full post. The `Link` component from Next.js is used to handle navigation.

There are many ways to customize the blog card, such as adding tags, author information, or publication date. You can extend the component by adding more props and styling as needed.

### Parallax Scroll Card

```jsx
import React, { useEffect, useRef, useState } from "react";
import Link from "next/link";
const ParallaxCard = ({ title, description, imageUrl, link, readTime }) => {
  const cardRef = useRef(null);
  const [offset, setOffset] = useState(0);

  useEffect(() => {
    const handleScroll = () => {
      if (cardRef.current) {
        const rect = cardRef.current.getBoundingClientRect();
        const scrolled = window.pageYOffset;
        const parallax = scrolled * 0.5;
        setOffset(parallax);
      }
    };

    window.addEventListener("scroll", handleScroll);
    return () => window.removeEventListener("scroll", handleScroll);
  }, []);

  return (
    <div
      ref={cardRef}
      className="max-w-sm rounded-2xl overflow-hidden shadow-2xl bg-white relative"
    >
      <div className="h-48 overflow-hidden">
        <img
          className="w-full h-full object-cover transition-transform duration-300"
          style={{ transform: `translateY(${offset * 0.1}px)` }}
          src={imageUrl}
          alt={title}
        />
        <div className="absolute top-4 right-4 bg-black/70 text-white px-2 py-1 rounded-full text-xs">
          {readTime} min read
        </div>
      </div>

      <div className="p-6">
        <h3 className="font-bold text-xl mb-3 text-gray-800">{title}</h3>
        <p className="text-gray-600 text-sm mb-6 leading-relaxed">
          {description}
        </p>

        <Link href={link}>
          <a className="group flex items-center text-blue-600 hover:text-blue-800 font-medium">
            Read Article

              className="w-4 h-4 ml-2 transition-transform group-hover:translate-x-1"
```

### Card with Audio Player

```jsx
import React, { useState, useRef } from "react";
import Link from "next/link";
const AudioCard = ({
  title,
  description,
  imageUrl,
  link,
  audioUrl,
  duration,
}) => {
  const [isPlaying, setIsPlaying] = useState(false);
  const [currentTime, setCurrentTime] = useState(0);
  const audioRef = useRef(null);

  const togglePlay = () => {
    if (audioRef.current) {
      if (isPlaying) {
        audioRef.current.pause();
      } else {
        audioRef.current.play();
      }
      setIsPlaying(!isPlaying);
    }
  };

  const handleTimeUpdate = () => {
    if (audioRef.current) {
      setCurrentTime(audioRef.current.currentTime);
    }
  };

  const formatTime = (time) => {
    const minutes = Math.floor(time / 60);
    const seconds = Math.floor(time % 60);
    return `${minutes}:${seconds.toString().padStart(2, "0")}`;
  };

  return (
    <div className="max-w-sm bg-white rounded-2xl shadow-xl overflow-hidden">
      <div className="relative">
        <img className="w-full h-48 object-cover" src={imageUrl} alt={title} />
        <div className="absolute inset-0 bg-black/50 flex items-center justify-center">
          <button
            onClick={togglePlay}
            className="bg-white/90 hover:bg-white text-black rounded-full p-4 transition-all hover:scale-110"
          >
            {isPlaying ? (
              <svg className="w-8 h-8" fill="currentColor" viewBox="0 0 24 24">
                <path d="M6 19h4V5H6v14zm8-14v14h4V5h-4z" />
              </svg>
            ) : (
              <svg className="w-8 h-8" fill="currentColor" viewBox="0 0 24 24">
                <path d="M8 5v14l11-7z" />
              </svg>
            )}
          </button>
        </div>
      </div>

      <div className="p-6">
        <h3 className="font-bold text-xl mb-2 text-gray-800">{title}</h3>
        <p className="text-gray-600 text-sm mb-4">{description}</p>

        <div className="bg-gray-100 rounded-lg p-4 mb-4">
          <div className="flex items-center justify-between text-sm text-gray-600 mb-2">
            <span>{formatTime(currentTime)}</span>
            <span>{duration}</span>
          </div>
          <div className="w-full bg-gray-300 rounded-full h-2">
            <div
              className="bg-blue-500 h-2 rounded-full transition-all"
              style={{
                width: `${(currentTime / (parseInt(duration) * 60)) * 100}%`,
              }}
            ></div>
          </div>
        </div>

        <Link href={link}>
          <a className="block text-center bg-blue-600 text-white py-3 rounded-lg hover:bg-blue-700 transition">
            Read Full Article
          </a>
        </Link>
      </div>

      <audio
        ref={audioRef}
        src={audioUrl}
        onTimeUpdate={handleTimeUpdate}
        onEnded={() => setIsPlaying(false)}
      />
    </div>
  );
};
export default AudioCard;
```
