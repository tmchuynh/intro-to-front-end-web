# Blog Details Dialog

## Table of Contents

## Dialogs

The `shadcn/ui` library provides a set of components that can be used to create dialogs. The `Dialog` component is the main component that you will use to create a dialog. It has several sub-components that you can use to customize the dialog.

### Dialog Components

- `Dialog`: The main component that wraps the dialog.
- `DialogTrigger`: The component that triggers the dialog to open.
- `DialogContent`: The component that contains the content of the dialog.
- `DialogHeader`: The component that contains the header of the dialog.
- `DialogFooter`: The component that contains the footer of the dialog.

#### Example

```tsx
// components/BlogDetailsDialog.tsx
import {
  Dialog,
  DialogTrigger,
  DialogContent,
  DialogHeader,
  DialogFooter,
} from "@/components/ui/dialog";
import { Button } from "@/components/ui/button";
import { useState } from "react";

// Simple dialog component for displaying blog post details in a modal overlay
export default function BlogDetailsDialog() {
  const [open, setOpen] = useState(false);

  return (
    <Dialog open={open} onOpenChange={setOpen}>
      <DialogTrigger asChild>
        <Button>Open Dialog</Button>
      </DialogTrigger>
      <DialogContent>
        <DialogHeader>
          <h2>Blog Details</h2>
        </DialogHeader>
        <p>This is the content of the blog details dialog.</p>
        <DialogFooter>
          <Button onClick={() => setOpen(false)}>Close</Button>
        </DialogFooter>
      </DialogContent>
    </Dialog>
  );
}
```

##### Usage

You can use the `BlogDetailsDialog` component in your blog post page to display the details of a blog post when the user clicks on a button or link. The dialog will open and display the content of the blog post.

```tsx
// pages/BlogPostPage.tsx
import BlogDetailsDialog from "@/components/BlogDetailsDialog";

// Blog post page component that displays a single blog post with dialog functionality
export default function BlogPostPage() {
  return (
    <div>
      <h1>Blog Post Title</h1>
      <p>This is the content of the blog post.</p>
      <BlogDetailsDialog />
      <BackToTop />
    </div>
  );
}
```

```tsx
import {
  Dialog,
  DialogTrigger,
  DialogContent,
  DialogHeader,
  DialogTitle,
  DialogDescription,
} from "@/components/ui/dialog";
import { Button } from "@/components/ui/button";
import { Progress } from "@/components/ui/progress";
import { ArrowRight, Eye, Clock, TrendingUp } from "lucide-react";
import { useState, useEffect } from "react";
import Image from "next/image";
import Link from "next/link";

// Progressive reading experience dialog with navigation and reading progress tracking
export default function BlogNavigationDialog({
  title,
  content,
  excerpt,
  imageUrl,
  slug,
  readTime,
  views,
  trending = false,
}) {
  const [open, setOpen] = useState(false);
  const [readingProgress, setReadingProgress] = useState(0);
  const [isNavigating, setIsNavigating] = useState(false);

  // @collapse
  useEffect(() => {
    if (open) {
      const timer = setInterval(() => {
        setReadingProgress((prev) => {
          if (prev >= 100) {
            clearInterval(timer);
            return 100;
          }
          return prev + 2;
        });
      }, 50);
      return () => clearInterval(timer);
    } else {
      setReadingProgress(0);
    }
  }, [open]);

  const handleNavigate = () => {
    setIsNavigating(true);
    // Simulate navigation delay
    setTimeout(() => {
      window.location.href = `/blog/${slug}`;
    }, 800);
  };

  return (
    <Dialog open={open} onOpenChange={setOpen}>
      <DialogTrigger asChild>
        <Button variant="default" className="group relative overflow-hidden">
          <span className="relative z-10 flex items-center gap-2">

          </span>
          <div className="absolute inset-0 bg-gradient-to-r from-blue-600 to-purple-600 opacity-0 group-hover:opacity-100 transition-opacity duration-300" />

          <div className="flex items-start justify-between">
            <div className="flex-1">

              <span>Reading Progress</span>
              <span>{readingProgress}%</span>

            <div className="relative rounded-xl overflow-hidden">
              <Image

                height={400}
                className="w-full h-64 object-cover"

              <div className="absolute top-4 right-4 bg-black/70 text-white px-3 py-1 rounded-full text-sm">
```

```tsx
// components/MultiStepBlogDialog.tsx
import {
  Dialog,
  DialogTrigger,
  DialogContent,
  DialogHeader,
  DialogTitle,
} from "@/components/ui/dialog";
import { Button } from "@/components/ui/button";
import { Progress } from "@/components/ui/progress";
import {
  ChevronLeft,
  ChevronRight,
  Bookmark,
  Share2,
  ThumbsUp,
} from "lucide-react";
import { useState } from "react";
import Image from "next/image";

// Step-by-step tutorial preview dialog for multi-section content navigation
export default function MultiStepBlogDialog({
  title,
  sections,
  imageUrl,
  author,
  readTime,
}) {
  const [open, setOpen] = useState(false);
  const [currentStep, setCurrentStep] = useState(0);
  const progress = ((currentStep + 1) / sections.length) * 100;

  const nextStep = () => {
    if (currentStep < sections.length - 1) {
      setCurrentStep(currentStep + 1);
    }
  };

  const prevStep = () => {
    if (currentStep > 0) {
      setCurrentStep(currentStep - 1);
    }
  };

  return (
    <Dialog open={open} onOpenChange={setOpen}>
      <DialogTrigger asChild>
        <Button variant="outline" className="group">
          <span>Step-by-Step Preview</span>
          <ChevronRight className="ml-2 h-4 w-4 transition-transform group-hover:translate-x-2" />
        </Button>
      </DialogTrigger>
      <DialogContent className="max-w-3xl">
        <DialogHeader>
          <DialogTitle className="text-xl">{title}</DialogTitle>
          <div className="flex items-center gap-4 text-sm text-gray-500">
            <span>
              Step {currentStep + 1} of {sections.length}
            </span>
            <span>‚Ä¢</span>
            <span>{readTime} min read</span>
            <span>‚Ä¢</span>
            <span>By {author}</span>
          </div>
        </DialogHeader>

        <div className="space-y-4">
          <Progress value={progress} className="h-2" />

          <div className="min-h-[400px] space-y-4">
            {imageUrl && currentStep === 0 && (

                src={imageUrl}
                alt={title}
                width={600}
                height={300}
                className="w-full h-48 object-cover rounded-lg"

            )}

              <h3 className="text-lg font-semibold">
                {sections[currentStep]?.title}
              </h3>
              <p className="text-gray-700 leading-relaxed">
                {sections[currentStep]?.content}
              </p>

                <div className="bg-gray-900 text-gray-100 p-4 rounded-lg overflow-x-auto">
                  <pre>
                    <code>{sections[currentStep].codeExample}</code>

                </div>

            </div>

            <div className="flex gap-2">
              <Button
                variant="ghost"
                size="sm"
                onClick={prevStep}
                disabled={currentStep === 0}
              >
                <ChevronLeft className="h-4 w-4 mr-1" />
                Previous
              </Button>

                variant="ghost"
                size="sm"
                onClick={nextStep}

                Next
                <ChevronRight className="h-4 w-4 ml-1" />

                <Share2 className="h-4 w-4" />

                <ThumbsUp className="h-4 w-4" />
              </Button>
            </div>
          </div>
        </div>
      </DialogContent>
```

## Alert Dialogs VS Regular Dialogs

Understanding the difference between Alert Dialogs and Regular Dialogs is crucial for creating effective user interfaces. Each serves different purposes and provides distinct user experiences.

### What are Alert Dialogs?

Alert Dialogs are modal components specifically designed for critical interactions that require immediate user attention. They typically interrupt the user's workflow to present important information, confirmations, or warnings.

#### Key Characteristics of Alert Dialogs:

- **Modal and Blocking**: They prevent interaction with the rest of the application
- **Focus Management**: Automatically trap focus within the dialog
- **Accessibility**: Built-in ARIA attributes for screen readers
- **Simplified Actions**: Usually limited to 1-3 action buttons
- **No Dismiss on Outside Click**: Generally can't be closed by clicking outside

### What are Regular Dialogs?

Regular Dialogs are more flexible modal components that can display various types of content, from simple messages to complex forms and interactive elements.

#### Key Characteristics of Regular Dialogs:

- **Flexible Content**: Can contain any type of content or components
- **Customizable Behavior**: More control over dismiss actions and interactions
- **Rich Interactions**: Support for forms, media, navigation, etc.
- **Variable Sizing**: Can be resized or have custom dimensions
- **Multiple Dismiss Options**: Can be closed via buttons, outside clicks, or escape key

## Comparison: Alert Dialogs VS Regular Dialogs

### Alert Dialogs

#### ‚úÖ Advantages

- **Clear Intent**: Designed specifically for critical actions and confirmations
- **Accessibility First**: Built-in ARIA roles and keyboard navigation
- **Consistent UX**: Standardized patterns users expect for confirmations
- **Lightweight**: Minimal API surface and simple implementation
- **Focus Management**: Automatic focus trapping and restoration
- **Semantic HTML**: Proper dialog role attributes for assistive technologies

##### ‚ùå Disadvantages

- **Limited Flexibility**: Restricted content types and layouts
- **Basic Styling**: Less customization options for visual design
- **Simple Actions**: Usually limited to confirmation/cancellation patterns
- **No Rich Content**: Cannot easily embed complex components or media
- **Fixed Behavior**: Less control over interaction patterns

- **Destructive Actions**: Confirming deletions, permanent changes
- **Error Messages**: Critical system errors requiring acknowledgment
- **Data Loss Prevention**: Warning before losing unsaved work
- **Permission Requests**: Asking for user consent or authorization
- **System Notifications**: Important system-wide announcements
- **Confirmation Actions**: Validating high-impact user decisions

## Decision Matrix: When to Use Which

### Use Alert Dialog When:

| Scenario                 | Why Alert Dialog                                        |
| ------------------------ | ------------------------------------------------------- |
| **Confirming Deletions** | Users need clear warning about irreversible actions     |

| **Permission Requests**  | Clear consent flows for sensitive operations            |
| **Account Actions**      | Confirming account-level changes (deletion, suspension) |
| **Logout Confirmations** | Ensuring users don't accidentally lose their session    |

### Use Regular Dialog When:

| Scenario               | Why Regular Dialog                                         |
| ---------------------- | ---------------------------------------------------------- |
| **Content Previews**   | Rich content display with images, formatting, interactions |
| **Form Interfaces**    | Complex forms with multiple fields and validation          |
| **Media Players**      | Video/audio content requiring custom controls              |
| **Step-by-Step Flows** | Multi-step processes with navigation                       |
| **Settings Panels**    | Configuration interfaces with multiple options             |
| **Help Documentation** | Detailed information with formatting and links             |

## Best Practices Summary

### For Alert Dialogs:

1. **Keep titles concise** but descriptive
2. **Use clear, action-oriented language** in buttons
3. **Provide sufficient context** in descriptions
4. **Use appropriate visual indicators** (icons, colors)
5. **Test with screen readers** to ensure accessibility

### For Regular Dialogs:

1. **Implement proper focus management** manually
2. **Add appropriate ARIA attributes** for accessibility
3. **Keep content scannable** with clear hierarchy
4. **Provide multiple ways to close** the dialog
5. **Optimize for mobile** responsive design
6. **Consider performance** with complex content

### Universal Guidelines:

- **Always provide a way to cancel** or go back
- **Use consistent language** across your application
- **Test keyboard navigation** thoroughly
- **Consider mobile users** with touch interfaces
- **Provide feedback** for user actions (loading states, success messages)

## Advanced Pattern: Conditional Dialog Type

Sometimes you might want to dynamically choose between alert and regular dialogs based on the context:

````tsx
```tsx
// components/ConditionalDialog.tsx
import { useState } from "react";
import {
  AlertDialog,
  AlertDialogAction,
  AlertDialogCancel,
  AlertDialogContent,
  AlertDialogDescription,
  AlertDialogFooter,
  AlertDialogHeader,
  AlertDialogTitle,
  AlertDialogTrigger,
} from "@/components/ui/alert-dialog";
import {
  Dialog,
  DialogContent,
  DialogHeader,
  DialogTitle,
  DialogTrigger,
} from "@/components/ui/dialog";

interface ConditionalDialogProps {
  mode: "alert" | "regular";
  trigger: React.ReactNode;
  title: string;
  description?: string;
  children?: React.ReactNode;
  onConfirm?: () => void;
  onCancel?: () => void;
}

// Conditional dialog component that dynamically switches between alert and regular dialog types
export function ConditionalDialog({
  mode,
  trigger,
  title,
  description,
  children,
  onConfirm,
  onCancel,
}: ConditionalDialogProps) {
  const [open, setOpen] = useState(false);

  if (mode === "alert") {
    return (
      <AlertDialog open={open} onOpenChange={setOpen}>
        <AlertDialogTrigger asChild>{trigger}</AlertDialogTrigger>
        <AlertDialogContent>
          <AlertDialogHeader>
            <AlertDialogTitle>{title}</AlertDialogTitle>
            {description && (
              <AlertDialogDescription>{description}</AlertDialogDescription>
            )}
          </AlertDialogHeader>
          <AlertDialogFooter>
            <AlertDialogCancel onClick={onCancel}>Cancel</AlertDialogCancel>
            <AlertDialogAction onClick={onConfirm}>Confirm</AlertDialogAction>
          </AlertDialogFooter>
        </AlertDialogContent>
      </AlertDialog>
    );
  }

  return (
    <Dialog open={open} onOpenChange={setOpen}>
      <DialogTrigger asChild>{trigger}</DialogTrigger>
      <DialogContent>
        <DialogHeader>
          <DialogTitle>{title}</DialogTitle>
        </DialogHeader>
        {children}
      </DialogContent>
    </Dialog>
  );
}

// Usage example component that demonstrates conditional dialog type selection
function BlogPostActions({ post, hasUnsavedChanges }) {
  return (
    <div className="flex gap-2">
      <ConditionalDialog
        mode={hasUnsavedChanges ? "alert" : "regular"}
        trigger={<Button>Delete Post</Button>}
        title={hasUnsavedChanges ? "Unsaved Changes" : "Delete Post"}
        description={
          hasUnsavedChanges
            ? "You have unsaved changes. Are you sure you want to delete this post?"
            : undefined
        }
        onConfirm={() => deletePost(post.id)}
      >
        {!hasUnsavedChanges && (
          <div className="space-y-4">
            <p>Are you sure you want to delete "{post.title}"?</p>
            <div className="flex justify-end gap-2">
              <Button variant="outline" onClick={() => setOpen(false)}>
                Cancel
              </Button>
              <Button variant="destructive" onClick={() => deletePost(post.id)}>
                Delete
              </Button>
            </div>
          </div>
        )}
      </ConditionalDialog>
    </div>
  );
}
````

This pattern allows you to adapt the dialog type based on the current application state or user context, providing the most appropriate user experience for each situation.
