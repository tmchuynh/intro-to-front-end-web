## Overview

There are multiple components that are used to handle navigation in the blog website. These components are designed to work together to provide a seamless navigation experience.

For example, the `Navigation` component is used to render the navigation bar, while the `NavigationLink` component is used to render individual links in the navigation bar. The `NavigationDropdown` component is used to render dropdown menus in the navigation bar.

In the example regarding the `layout.tsx` file, a sidebar is used alongside a top navbar to provide a  navigation experience. The sidebar contains links to different sections of the website, while the top navbar can include additional controls or branding.

Below are some examples of sidebars, top navbars, and other navigation components that can be used in the blog website.

## A Sidebar with Dropdowns

This is a sidebar navigation that includes dropdown menus for sub-sections. It can be used to organize content into categories and subcategories. Sidebars are particularly useful for websites with a lot of content, as they allow users to easily navigate through different sections. They are typically placed on the left side of the page, but can also be positioned on the right side.

```tsx
"use client";
import Link from "next/link";
import { useState } from "react";
import { ChevronDownIcon, ChevronRightIcon } from "@heroicons/react/24/outline";

export default function SidebarWithDropdowns() {
  const [openDropdowns, setOpenDropdowns] = useState<string[]>([]);

  const toggleDropdown = (section: string) => {
    setOpenDropdowns((prev) =>
      prev.includes(section)
        ? prev.filter((item) => item !== section)
        : [...prev, section],
    );
  };

  return (
    <aside className="w-64 bg-gray-50 dark:bg-gray-800 min-h-screen border-r border-gray-200 dark:border-gray-700">
      <div className="p-4">
        <h2 className="text-lg font-semibold text-gray-900 dark:text-white mb-4">
          Navigation
        </h2>

        <nav className="space-y-1">
          {/* Simple Link */}
          <Link
            href="/dashboard"
            className="flex items-center px-3 py-2 text-sm font-medium text-gray-700 hover:bg-gray-100 dark:text-gray-300 dark:hover:bg-gray-700 rounded-md transition-colors"
          >
            Dashboard
          </Link>

          {/* Dropdown Section - Getting Started */}
          <div>
            <button
              onClick={() => toggleDropdown("getting-started")}
              className="flex items-center justify-between w-full px-3 py-2 text-sm font-medium text-gray-700 hover:bg-gray-100 dark:text-gray-300 dark:hover:bg-gray-700 rounded-md transition-colors"
            >
              <span>Getting Started</span>
              {openDropdowns.includes("getting-started") ? (
                <ChevronDownIcon className="w-4 h-4" />
              ) : (
                <ChevronRightIcon className="w-4 h-4" />
              )}
            </button>

            {openDropdowns.includes("getting-started") && (
              <div className="ml-4 mt-1 space-y-1">
                <Link
                  href="/getting-started/installation"
                  className="block px-3 py-2 text-sm text-gray-600 hover:bg-gray-100 dark:text-gray-400 dark:hover:bg-gray-700 rounded-md transition-colors"
                >
                  Installation
                </Link>
                <Link
                  href="/getting-started/configuration"
                  className="block px-3 py-2 text-sm text-gray-600 hover:bg-gray-100 dark:text-gray-400 dark:hover:bg-gray-700 rounded-md transition-colors"
                >
                  Configuration
                </Link>
                <Link
                  href="/getting-started/first-steps"
                  className="block px-3 py-2 text-sm text-gray-600 hover:bg-gray-100 dark:text-gray-400 dark:hover:bg-gray-700 rounded-md transition-colors"
                >
                  First Steps
                </Link>
              </div>
            )}
          </div>

          {/* Dropdown Section - Components */}
          <div>
            <button
              onClick={() => toggleDropdown("components")}
              className="flex items-center justify-between w-full px-3 py-2 text-sm font-medium text-gray-700 hover:bg-gray-100 dark:text-gray-300 dark:hover:bg-gray-700 rounded-md transition-colors"
            >
              <span>Components</span>
              {openDropdowns.includes("components") ? (
                <ChevronDownIcon className="w-4 h-4" />
              ) : (
                <ChevronRightIcon className="w-4 h-4" />
              )}
            </button>

            {openDropdowns.includes("components") && (
              <div className="ml-4 mt-1 space-y-1">
                <Link
                  href="/components/buttons"
                  className="block px-3 py-2 text-sm text-gray-600 hover:bg-gray-100 dark:text-gray-400 dark:hover:bg-gray-700 rounded-md transition-colors"
                >
                  Buttons
                </Link>
                <Link
                  href="/components/forms"
                  className="block px-3 py-2 text-sm text-gray-600 hover:bg-gray-100 dark:text-gray-400 dark:hover:bg-gray-700 rounded-md transition-colors"
                >
                  Forms
                </Link>
                <Link
                  href="/components/navigation"
                  className="block px-3 py-2 text-sm text-gray-600 hover:bg-gray-100 dark:text-gray-400 dark:hover:bg-gray-700 rounded-md transition-colors"
                >
                  Navigation
                </Link>
              </div>
            )}
          </div>

          {/* Simple Link */}
          <Link
            href="/api-reference"
            className="flex items-center px-3 py-2 text-sm font-medium text-gray-700 hover:bg-gray-100 dark:text-gray-300 dark:hover:bg-gray-700 rounded-md transition-colors"
          >
            API Reference
          </Link>
        </nav>
      </div>
    </aside>
  );
}
```

### Customization Options

#### State Management

- **Open by Default**: Initialize `openDropdowns` with section names to start expanded
- **Single Dropdown**: Modify state logic to only allow one dropdown open at a time
- **Persist State**: Use localStorage to remember which dropdowns were open

##### Styling Modifications

- **Width**: Change `w-64` to `w-48`, `w-72`, or `w-80` for different sidebar widths
- **Background**: Modify `bg-gray-50` for different background colors
- **Icons**: Replace Heroicons with your preferred icon library (Lucide, Feather, etc.)
- **Animations**: Add `transition-all duration-200` for smooth open/close animations

###### Layout Adjustments

- **Right Sidebar**: Add `ml-auto` and position on the right side of layout
- **Collapsible**: Add a toggle button to show/hide entire sidebar
- **Mobile Responsive**: Add mobile overlay with backdrop for smaller screens

###### Advanced Features to Add

- Search functionality within sidebar items
- Badge indicators for new or updated sections
- Nested multi-level dropdowns (3+ levels deep)
- Drag and drop reordering of navigation items

## A Top Navbar with a Search Bar

This is a top navigation bar that includes a search bar. It can be used to provide quick access to search functionality, allowing users to find content quickly. The search bar can be integrated with a search engine or a custom search function to provide relevant results based on user queries.

```tsx
                stroke="currentColor"
              >

                  strokeLinecap="round"
                  strokeLinejoin="round"
                  strokeWidth={2}
                  d="M4 6h16M4 12h16M4 18h16"
```

## Hamburger Menu

A hamburger menu is a compact navigation menu that is typically represented by three horizontal lines. It is commonly used in mobile applications and responsive designs to save space. When clicked, it expands to reveal the full navigation menu. Hamburger menus are useful for mobile devices where screen space is limited, allowing users to access the navigation options without cluttering the interface.

Typically, hamburger menus are used in conjunction with responsive design techniques to ensure that the navigation remains accessible across different screen sizes.


## Sticky Navigation

Sticky navigation remains fixed at the top of the viewport as the user scrolls down the page. It provides a persistent navigation experience, allowing users to access the navigation menu without having to scroll back to the top. Sticky navigation is particularly useful for long pages or articles, as it keeps the navigation options readily available without interrupting the reading experience.

```tsx
"use client";
import Link from "next/link";
import { useState, useEffect } from "react";

export default function StickyNavigation() {
  const [isScrolled, setIsScrolled] = useState(false);

    const handleScroll = () => {
      const scrollTop = window.scrollY;
      setIsScrolled(scrollTop > 10);
    };

    window.addEventListener("scroll", handleScroll);
    return () => window.removeEventListener("scroll", handleScroll);
  }, []);

  return (
    <nav
      className={`
        fixed top-0 left-0 right-0 z-50 transition-all duration-300 ease-in-out
        ${
          isScrolled
            ? "bg-white/95 dark:bg-gray-900/95 backdrop-blur-md shadow-lg border-b border-gray-200/50 dark:border-gray-700/50"
            : "bg-transparent"
        }

    >
      <div className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8">
        <div className="flex justify-between items-center h-16">
          {/* Logo */}
          <Link
            href="/"
            className={`
              text-xl font-bold transition-colors duration-300
              ${isScrolled ? "text-gray-900 dark:text-white" : "text-white"}
            `}
          >
            StickyBrand
          </Link>

          {/* Navigation Links */}
          <div className="hidden md:flex space-x-6">
            <Link
              href="/"
              className={`
                px-3 py-2 text-sm font-medium transition-all duration-300
                ${
                  isScrolled
                    ? "text-gray-700 hover:text-blue-600 dark:text-gray-300 dark:hover:text-blue-400"
                    : "text-white/90 hover:text-white"
                }
              `}
            >
              Home

            <Link
              href="/about"
              className={`
                px-3 py-2 text-sm font-medium transition-all duration-300
                ${
                  isScrolled
                    ? "text-gray-700 hover:text-blue-600 dark:text-gray-300 dark:hover:text-blue-400"

                }
              `}

                px-3 py-2 text-sm font-medium transition-all duration-300
```

## Breadcrumb Navigation

Breadcrumb navigation provides a trail of links that shows the user's current location within the website's hierarchy. It allows users to easily navigate back to previous sections or the homepage. Breadcrumbs are especially useful for websites with deep content structures, as they provide context and help users understand their position within the site.

```tsx
import Link from "next/link";
import { ChevronRightIcon, HomeIcon } from "@heroicons/react/24/outline";

interface BreadcrumbItem {
  label: string;
  href?: string;
  current?: boolean;
}

interface BreadcrumbNavigationProps {
  items: BreadcrumbItem[];
  showHome?: boolean;
}

export default function BreadcrumbNavigation({
  items,
  showHome = true,
}: BreadcrumbNavigationProps) {
  return (
    <nav className="flex items-center space-x-2 text-sm text-gray-500 dark:text-gray-400 py-4">
      {/* Home Link */}
      {showHome && (
        <>
          <Link
            href="/"
            className="flex items-center hover:text-gray-700 dark:hover:text-gray-200 transition-colors"
          >
            <HomeIcon className="w-4 h-4 mr-1" />
            <span className="sr-only">Home</span>
          </Link>
          {items.length > 0 && (
            <ChevronRightIcon className="w-4 h-4 text-gray-400" />
          )}
        </>
      )}

      {/* Breadcrumb Items */}
      {items.map((item, index) => (
        <div key={index} className="flex items-center space-x-2">
          {item.href && !item.current ? (
            <Link
              href={item.href}
              className="hover:text-gray-700 dark:hover:text-gray-200 transition-colors"
            >
              {item.label}
            </Link>
          ) : (
            <span
              className={`${
                item.current
                  ? "font-medium text-gray-900 dark:text-white"
                  : "text-gray-500 dark:text-gray-400"
              }`}
            >
              {item.label}
            </span>
          )}

          {/* Separator */}
          {index < items.length - 1 && (
            <ChevronRightIcon className="w-4 h-4 text-gray-400" />
          )}
        </div>
      ))}
    </nav>
  );
}

// Example Usage Component
```

## Dynamic Breadcrumbs

Dynamic breadcrumbs automatically update based on the user's current location within the website. They can be generated based on the URL structure or the content hierarchy, providing a real-time navigation aid. Dynamic breadcrumbs enhance user experience by ensuring that users always have access to relevant navigation links, regardless of where they are on the site.

```tsx
"use client";
import Link from "next/link";
import { usePathname } from "next/navigation";
import { ChevronRightIcon, HomeIcon } from "@heroicons/react/24/outline";
import { useMemo } from "react";

interface RouteMapping {
  [key: string]: {
    label: string;
    icon?: React.ComponentType<{ className?: string }>;
  };
}

// Define custom labels for routes
const routeMapping: RouteMapping = {
  blog: { label: "Blog" },
  products: { label: "Products" },
  electronics: { label: "Electronics" },
  smartphones: { label: "Smartphones" },
  laptops: { label: "Laptops" },
  categories: { label: "Categories" },
  about: { label: "About Us" },
  contact: { label: "Contact" },
  support: { label: "Support" },
  documentation: { label: "Documentation" },
  api: { label: "API Reference" },
  "getting-started": { label: "Getting Started" },
  tutorials: { label: "Tutorials" },
};

interface DynamicBreadcrumbsProps {
  homeLabel?: string;
  maxItems?: number;
  showHome?: boolean;
  customRoutes?: RouteMapping;
}

export default function DynamicBreadcrumbs({
  homeLabel = "Home",
  maxItems = 5,
  showHome = true,
  customRoutes = {},
}: DynamicBreadcrumbsProps) {
  const pathname = usePathname();

  const breadcrumbs = useMemo(() => {
    // Combine default routes with custom routes
    const allRoutes = { ...routeMapping, ...customRoutes };

    // Split pathname and filter out empty strings
    const pathSegments = pathname
      .split("/")
      .filter((segment) => segment !== "");

    // If we're on home page, return empty array
    if (pathSegments.length === 0) {
      return [];
    }

    // Build breadcrumb items
    // @collapse
    const items = pathSegments.map((segment, index) => {
      const href = "/" + pathSegments.slice(0, index + 1).join("/");
      const isLast = index === pathSegments.length - 1;

      // Get custom label or format the segment
      const label =
        allRoutes[segment]?.label ||
        segment
          .split("-")
          .map((word) => word.charAt(0).toUpperCase() + word.slice(1))
          .join(" ");

      return {
        label,
        href: isLast ? undefined : href,
        current: isLast,
        icon: allRoutes[segment]?.icon,
      };
    });

    // Limit items if maxItems is set
    if (items.length > maxItems) {
      return [
        ...items.slice(0, 1), // Keep first item
        { label: "...", ellipsis: true },
        ...items.slice(-(maxItems - 2)), // Keep last items
      ];
    }

    return items;
  }, [pathname, maxItems, customRoutes]);

  return (
    <nav
      className="flex items-center space-x-2 text-sm text-gray-500 dark:text-gray-400 py-4"
      aria-label="Breadcrumb"
    >
      {/* Home Link */}
      {showHome && (
        <>
          <Link
            href="/"
            className="flex items-center hover:text-gray-700 dark:hover:text-gray-200 transition-colors group"
          >
            <HomeIcon className="w-4 h-4 mr-1 group-hover:scale-110 transition-transform" />
            <span className="hidden sm:inline">{homeLabel}</span>
          </Link>
          {breadcrumbs.length > 0 && (
            <ChevronRightIcon className="w-4 h-4 text-gray-400 flex-shrink-0" />
          )}
        </>
      )}

      {/* Dynamic Breadcrumb Items */}
      {breadcrumbs.map((item, index) => (
        <div key={index} className="flex items-center space-x-2">
          {item.ellipsis ? (
            <span className="text-gray-400 px-2">...</span>
          ) : (
            <>
              {item.href ? (
                <Link
                  href={item.href}
                  className="hover:text-gray-700 dark:hover:text-gray-200 transition-colors flex items-center"
                >
                  {item.icon && <item.icon className="w-4 h-4 mr-1" />}
                  <span className="truncate max-w-[120px] sm:max-w-none">
                    {item.label}
                  </span>
                </Link>
              ) : (
                <span
                  className={`font-medium text-gray-900 dark:text-white flex items-center ${
                    item.current ? "text-blue-600 dark:text-blue-400" : ""
                  }`}
                >
                  {item.icon && <item.icon className="w-4 h-4 mr-1" />}
                  <span className="truncate max-w-[120px] sm:max-w-none">
                    {item.label}
                  </span>
                </span>
              )}
            </>
          )}
```
