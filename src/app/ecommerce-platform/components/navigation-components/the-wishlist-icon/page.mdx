## The `WishlistIcon` Component

```tsx
// src/components/navigation/WishlistIcon.tsx
import { useAuth } from "@/app/context/authContext";
import { useWishlist } from "@/app/context/wishlistContext";
import { Button } from "@/components/ui/button";
import { Heart } from "lucide-react";
import Link from "next/link";

export default function WishlistIcon() {
  const { wishlistCount } = useWishlist();
  const { isLoggedIn } = useAuth();

  if (!isLoggedIn) {
    return null;
  }

  return (
    <Link href="/wishlist">
      <Button variant="ghost" size="sm" className="relative p-2">
        <Heart className="w-5 h-5" />
        {wishlistCount > 0 && (
          <span className="-top-1 -right-1 absolute flex justify-center items-center bg-red-500 rounded-full w-5 min-w-[20px] h-5 text-white text-xs">
            {wishlistCount > 99 ? "99+" : wishlistCount}
          </span>
        )}
        <span className="sr-only">Wishlist</span>
      </Button>
    </Link>
  );
}
```

### Explanation of the Code

This foundational WishlistIcon component demonstrates intelligent component architecture through strategic context integration and conditional rendering patterns:

#### Context Integration Excellence:

- **useWishlist Hook**: Extracts `wishlistCount` from global wishlist context, enabling real-time synchronization across the application
- **useAuth Hook**: Provides `isLoggedIn` status for authentication-aware UI management and privacy protection

#### Privacy-First Design Pattern:

- **Conditional Rendering**: Returns `null` for unauthenticated users, completely hiding the wishlist functionality
- **Security by Default**: Prevents unauthorized access to wishlist features while maintaining clean component architecture
- **Performance Benefits**: Eliminates unnecessary DOM rendering for users who cannot access wishlist features

#### Navigation and Visual Hierarchy:

- **Next.js Link Wrapper**: Provides optimized client-side navigation with automatic prefetching for enhanced performance
- **Ghost Button Variant**: Minimalist design integrates seamlessly with navigation while maintaining clear interaction affordances
- **Relative Positioning**: Enables precise badge placement for optimal visual hierarchy and user attention management

#### Advanced Badge System:

- **Conditional Badge Display**: Only renders count when items exist, preventing visual noise for empty wishlists
- **Smart Number Formatting**: Displays "99+" for quantities over 99 to prevent layout breaking while maintaining readability
- **Strategic Positioning**: Uses absolute positioning (-top-1 -right-1) following established UI patterns for notification badges

#### Accessibility Implementation:

- **Screen Reader Support**: Hidden span with "Wishlist" text provides essential context for assistive technologies

// Advanced wishlist interaction tracking
const useAdvancedWishlistIcon = () => {
  const { wishlistCount, wishlistValue, recentlyAdded } = useWishlist();
  const { user, isLoggedIn } = useAuth();
  const { trackEvent } = useAnalytics();

  const handleWishlistClick = useCallback(() => {

      trackEvent("wishlist_icon_clicked", {
        user_id: user?.id,
        wishlist_count: wishlistCount,
        wishlist_value: wishlistValue,
        recently_added_items: recentlyAdded?.length || 0,
        user_tier: user?.tier,
        session_duration: getSessionDuration(),
      });
    }
  }, [
    isLoggedIn,
    user,
    wishlistCount,
    wishlistValue,
    recentlyAdded,
    trackEvent,
  ]);

  const wishlistStatus = useMemo(() => {
    if (wishlistCount === 0) return "empty";
    if (wishlistCount <= 5) return "growing";
    if (wishlistCount <= 20) return "active";
    return "collector";
  }, [wishlistCount]);

  const shouldShowPulse = useMemo(() => {
    return (
      recentlyAdded &&
      recentlyAdded.some((item) => Date.now() - item.addedAt.getTime() < 3000)
    );
  }, [recentlyAdded]);

  return {
    wishlistCount,
    wishlistValue,
    handleWishlistClick,
    wishlistStatus,
    shouldShowPulse,
    isLoggedIn,
  };
};
