      date: deliveredDate,
      status: "Delivered",
      location: "Destination",
      description: "Package has been successfully delivered.",

    order,
    events: events.sort((a, b) => b.date.getTime() - a.date.getTime()),

    eventStream.onmessage = (event) => {
      const orderEvent = JSON.parse(event.data) as OrderEvent;
      
      // Update order events
      setOrderEvents(prev => [orderEvent, ...prev.slice(0, 99)]);
      
      // Check for alert conditions
      const alert = evaluateAlertConditions(orderEvent);
      if (alert) {
        setAlerts(prev => [alert, ...prev]);
        triggerNotification(alert);
      }
    };

    return () => eventStream.close();
  }, []);

  const monitoringCapabilities = {
    // Real-time metrics
    getRealTimeMetrics: () => ({
      ordersPerMinute: calculateOrderVelocity(orderEvents),
      averageProcessingTime: calculateProcessingTime(orderEvents),
      errorRate: calculateErrorRate(orderEvents),
      customerSatisfactionTrend: calculateSatisfactionTrend(orderEvents)
    }),

    // Alert management
    acknowledgeAlert: (alertId: string) => {
      setAlerts(prev => prev.filter(alert => alert.id !== alertId));
    },

    // Performance monitoring
    getPerformanceMetrics: () => ({
      fulfillmentVelocity: calculateFulfillmentVelocity(),
      qualityMetrics: calculateQualityMetrics(),
      customerServiceMetrics: calculateServiceMetrics()
    })
  };

  return { ...baseHook, orderEvents, alerts, monitoring: monitoringCapabilities };
};
