## Handling Input Change and Submit

```tsx
// src/app/register/page.tsx
const handleInputChange = (e: React.ChangeEvent<HTMLInputElement>) => {
  setFormData({
    ...formData,
    [e.target.name]: e.target.value,
  });
};

const handleSubmit = async (e: React.FormEvent) => {
  e.preventDefault();

  if (formData.password !== formData.confirmPassword) {
    toast.error("Passwords do not match");
    return;
  }

  if (formData.password.length < 6) {
    toast.error("Password must be at least 6 characters long");
    return;
  }

  if (!formData.username.trim() || !formData.phoneNumber.trim()) {
    toast.error("Please fill in all required fields");
    return;
  }

  setIsLoading(true);

  const result = await register({
    username: formData.username.trim(),
    phoneNumber: formData.phoneNumber.trim(),
    email: formData.email.trim() || undefined,
    password: formData.password,
  });

  if (result.success) {
    toast.success(result.message);
    router.push("/membership");
  } else {
    toast.error(result.message);
  }

  setIsLoading(false);
};
```

### Explanation of the Code

This section handles the form submission and input changes. It checks for password confirmation, minimum password length, and required fields before calling the `register` function from the authentication context. If registration is successful, it redirects to the membership page; otherwise, it displays an error message.
