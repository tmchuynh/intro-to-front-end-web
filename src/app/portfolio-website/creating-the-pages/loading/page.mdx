# Loading Component

## Table of Contents

## About This Component

This component is used to display a loading state in the portfolio website. It is typically used when fetching data or performing asynchronous operations.

## Usage

You can use this loading component in your pages to indicate that content is being loaded. Below is an example of how to implement a simple loading component using React.

```tsx
import React, { useState } from "react";
import Loading from "@/components/Loading";
function MyComponent() {
  const [loading, setLoading] = useState(true);

  if (loading) {
    return <Loading />;
  }
  return (
    <div>
      {/* Your main content goes here */}
      <h1>Welcome to My Portfolio</h1>
      <p>This is the main content of the page.</p>
    </div>
  );
}
```

### Example #4: Dynamic Progress Bar with Real-time Feedback

An interactive progress bar loader with real-time progress tracking and dynamic messaging - ideal for file uploads, data processing, or multi-step operations.

```tsx
import React, { useState, useEffect } from "react";

const DynamicProgressLoader: React.FC = () => {
  const [progress, setProgress] = useState(0);
  const [currentStep, setCurrentStep] = useState(0);
  const [isComplete, setIsComplete] = useState(false);

  const loadingSteps = [
    { label: "Initializing portfolio", duration: 800 },
    { label: "Loading projects", duration: 1200 },
    { label: "Fetching images", duration: 1000 },
    { label: "Processing data", duration: 600 },
    { label: "Optimizing performance", duration: 400 },
    { label: "Finalizing", duration: 300 },
  ];

  useEffect(() => {
    let progressTimer: NodeJS.Timeout;
    let stepTimer: NodeJS.Timeout;

    const runProgress = () => {
      if (currentStep < loadingSteps.length) {
        const step = loadingSteps[currentStep];
        const stepProgress = 100 / loadingSteps.length;
        const startProgress = currentStep * stepProgress;
        const endProgress = (currentStep + 1) * stepProgress;

        let localProgress = 0;
        const increment = 2;
        const interval = step.duration / (stepProgress / increment);

        progressTimer = setInterval(() => {
          localProgress += increment;
          const currentProgress = startProgress + localProgress;

          if (currentProgress >= endProgress) {
            setProgress(endProgress);
            clearInterval(progressTimer);

            stepTimer = setTimeout(() => {
              setCurrentStep((prev) => prev + 1);
            }, 100);
          } else {
            setProgress(currentProgress);
          }
        }, interval);
      } else {
        setIsComplete(true);
      }
    };

    runProgress();

    return () => {
      clearInterval(progressTimer);
      clearTimeout(stepTimer);
    };
  }, [currentStep]);

  if (isComplete) {
    return (
      <div className="flex items-center justify-center min-h-screen bg-gradient-to-br from-green-50 to-blue-50">
        <div className="text-center space-y-4">
          <div className="w-16 h-16 bg-green-500 rounded-full flex items-center justify-center mx-auto animate-bounce">
            <span className="text-white text-2xl">✓</span>
          </div>
          <h2 className="text-2xl font-bold text-green-700">Ready!</h2>
          <p className="text-green-600">
            Your portfolio has loaded successfully
          </p>
        </div>
      </div>
    );
  }

  return (
    <div className="flex items-center justify-center min-h-screen bg-gradient-to-br from-slate-900 via-purple-900 to-slate-900 text-white">
      <div className="w-96 space-y-6 p-8">
        {/* Header */}
        <div className="text-center space-y-2">
          <h2 className="text-2xl font-bold bg-gradient-to-r from-blue-400 to-purple-400 bg-clip-text text-transparent">
            Loading Portfolio
          </h2>
          <p className="text-slate-400">
            Please wait while we prepare everything
          </p>
        </div>

        {/* Progress Circle */}
        <div className="flex justify-center">
          <div className="relative w-24 h-24">
            <svg
              className="w-24 h-24 transform -rotate-90"
              viewBox="0 0 100 100"
            >
              {/* Background circle */}
              <circle
                cx="50"
                cy="50"
                r="40"
                stroke="currentColor"
                strokeWidth="8"
                fill="none"
                className="text-slate-700"
              />
              {/* Progress circle */}
              <circle
                cx="50"
                cy="50"
                r="40"
                stroke="currentColor"
                strokeWidth="8"
                fill="none"
                strokeLinecap="round"
                className="text-blue-400"
                style={{
                  strokeDasharray: `${2 * Math.PI * 40}`,
                  strokeDashoffset: `${2 * Math.PI * 40 * (1 - progress / 100)}`,
                  transition: "stroke-dashoffset 0.3s ease",
                }}
              />
            </svg>
            <div className="absolute inset-0 flex items-center justify-center">
              <span className="text-lg font-bold">{Math.round(progress)}%</span>
            </div>
          </div>
        </div>

        {/* Progress Bar */}
        <div className="space-y-2">
          <div className="w-full bg-slate-700 rounded-full h-2 overflow-hidden">
            <div
              className="bg-gradient-to-r from-blue-500 to-purple-500 h-2 rounded-full transition-all duration-300 ease-out"
              style={{ width: `${progress}%` }}
            ></div>
          </div>
          <div className="flex justify-between text-sm text-slate-400">
            <span>0%</span>
            <span>100%</span>
          </div>
        </div>

        {/* Current Step */}
        <div className="space-y-4">
          <div className="text-center">
            <p className="text-blue-300 font-medium animate-pulse">
              {currentStep < loadingSteps.length
                ? loadingSteps[currentStep].label
                : "Complete"}
            </p>
          </div>

          {/* Step Indicators */}
          <div className="flex justify-between">
            {loadingSteps.map((step, index) => (
              <div key={index} className="flex flex-col items-center space-y-1">
                <div
                  className={`w-3 h-3 rounded-full transition-colors duration-300 ${
                    index < currentStep
                      ? "bg-green-400"
                      : index === currentStep
                        ? "bg-blue-400 animate-pulse"
                        : "bg-slate-600"
                  }`}
                ></div>
                <span className="text-xs text-slate-500 text-center max-w-16">
                  {step.label.split(" ")[0]}
                </span>
              </div>
            ))}
          </div>
        </div>

        {/* Additional Info */}
        <div className="bg-slate-800 rounded-lg p-4 text-sm">
          <div className="flex justify-between items-center">
            <span className="text-slate-400">Status:</span>
            <span className="text-blue-300">
              Step {currentStep + 1} of {loadingSteps.length}
            </span>
          </div>
          <div className="flex justify-between items-center mt-2">
            <span className="text-slate-400">Estimated time:</span>
            <span className="text-purple-300">
              ~{Math.max(0, 4 - currentStep)}s remaining
            </span>
          </div>
        </div>
      </div>
    </div>
  );
};
export default DynamicProgressLoader;
```

##### Customization Options

- **Steps Configuration**: Modify the `loadingSteps` array to match your app's actual loading process
- **Timing**: Adjust `duration` values for each step based on real loading times
- **Colors**: Change gradient colors from `blue-purple` to match your brand
- **Progress Indicators**: Switch between circular, linear, or both progress displays
- **Completion State**: Customize the success message and animation
- **Real-time Data**: Connect to actual loading events instead of simulated progress
- **Multiple Bars**: Add separate progress bars for different loading aspects
- **Sound Effects**: Include audio feedback for step completions
- **Error Handling**: Add error states for failed loading steps

### Example #5: Interactive Bouncing Balls Loader with Physics

A playful and engaging loader featuring animated bouncing balls with realistic physics, perfect for creative portfolios or fun applications.

```tsx
import React, { useEffect, useRef } from "react";

interface Ball {
  x: number;
  y: number;
  vx: number;
  vy: number;
  radius: number;
  color: string;
  bounce: number;
}

const InteractiveBounceLoader: React.FC = () => {
  const canvasRef = useRef<HTMLCanvasElement>(null);
  const animationRef = useRef<number>();
  const ballsRef = useRef<Ball[]>([]);

  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;

    const ctx = canvas.getContext("2d");
    if (!ctx) return;

    // Set canvas size
    const resizeCanvas = () => {
      canvas.width = canvas.offsetWidth * window.devicePixelRatio;
      canvas.height = canvas.offsetHeight * window.devicePixelRatio;
      ctx.scale(window.devicePixelRatio, window.devicePixelRatio);
    };

    resizeCanvas();
    window.addEventListener("resize", resizeCanvas);

    // Initialize balls
    const colors = [
      "#3B82F6",
      "#8B5CF6",
      "#EF4444",
      "#10B981",
      "#F59E0B",
      "#EC4899",
    ];
    ballsRef.current = Array.from({ length: 6 }, (_, i) => ({
      x: Math.random() * (canvas.offsetWidth - 40) + 20,
      y: Math.random() * (canvas.offsetHeight - 40) + 20,
      vx: (Math.random() - 0.5) * 8,
      vy: (Math.random() - 0.5) * 8,
      radius: Math.random() * 15 + 10,
      color: colors[i % colors.length],
      bounce: 0.85 + Math.random() * 0.1,
    }));

    // Animation loop
    const animate = () => {
      ctx.clearRect(0, 0, canvas.offsetWidth, canvas.offsetHeight);

      ballsRef.current.forEach((ball, index) => {
        // Update position
        ball.x += ball.vx;
        ball.y += ball.vy;

        // Add gravity
        ball.vy += 0.3;

        // Bounce off walls
        if (ball.x + ball.radius > canvas.offsetWidth) {
          ball.x = canvas.offsetWidth - ball.radius;
          ball.vx *= -ball.bounce;
        }
        if (ball.x - ball.radius < 0) {
          ball.x = ball.radius;
          ball.vx *= -ball.bounce;
        }
        if (ball.y + ball.radius > canvas.offsetHeight) {
          ball.y = canvas.offsetHeight - ball.radius;
          ball.vy *= -ball.bounce;
        }
        if (ball.y - ball.radius < 0) {
          ball.y = ball.radius;
          ball.vy *= -ball.bounce;
        }

        // Ball-to-ball collision
        ballsRef.current.forEach((otherBall, otherIndex) => {
          if (index !== otherIndex) {
            const dx = ball.x - otherBall.x;
            const dy = ball.y - otherBall.y;
            const distance = Math.sqrt(dx * dx + dy * dy);

            if (distance < ball.radius + otherBall.radius) {
              // Simple collision response
              const angle = Math.atan2(dy, dx);
              const targetX =
                otherBall.x +
                Math.cos(angle) * (ball.radius + otherBall.radius);
              const targetY =
                otherBall.y +
                Math.sin(angle) * (ball.radius + otherBall.radius);

              ball.x = targetX;
              ball.y = targetY;

              const tempVx = ball.vx;
              const tempVy = ball.vy;
              ball.vx = otherBall.vx * 0.8;
              ball.vy = otherBall.vy * 0.8;
              otherBall.vx = tempVx * 0.8;
              otherBall.vy = tempVy * 0.8;
            }
          }
        });

        // Draw ball with shadow and gradient
        ctx.save();

        // Shadow
        ctx.shadowColor = "rgba(0, 0, 0, 0.3)";
        ctx.shadowBlur = 10;
        ctx.shadowOffsetX = 3;
        ctx.shadowOffsetY = 3;

        // Gradient fill
        const gradient = ctx.createRadialGradient(
          ball.x - ball.radius * 0.3,
          ball.y - ball.radius * 0.3,
          0,
          ball.x,
          ball.y,
          ball.radius,
        );
        gradient.addColorStop(0, ball.color);
        gradient.addColorStop(1, ball.color + "88");

        ctx.beginPath();
        ctx.arc(ball.x, ball.y, ball.radius, 0, Math.PI * 2);
        ctx.fillStyle = gradient;
        ctx.fill();

        // Highlight
        ctx.shadowColor = "transparent";
        const highlight = ctx.createRadialGradient(
          ball.x - ball.radius * 0.4,
          ball.y - ball.radius * 0.4,
          0,
          ball.x - ball.radius * 0.4,
          ball.y - ball.radius * 0.4,
          ball.radius * 0.5,
        );
        highlight.addColorStop(0, "rgba(255, 255, 255, 0.8)");
        highlight.addColorStop(1, "rgba(255, 255, 255, 0)");

        ctx.beginPath();
        ctx.arc(ball.x, ball.y, ball.radius, 0, Math.PI * 2);
        ctx.fillStyle = highlight;
        ctx.fill();

        ctx.restore();
      });

      animationRef.current = requestAnimationFrame(animate);
    };

    animate();

    // Mouse interaction
    const handleMouseMove = (e: MouseEvent) => {
      const rect = canvas.getBoundingClientRect();
      const mouseX = e.clientX - rect.left;
      const mouseY = e.clientY - rect.top;

      ballsRef.current.forEach((ball) => {
        const dx = mouseX - ball.x;
        const dy = mouseY - ball.y;
        const distance = Math.sqrt(dx * dx + dy * dy);

        if (distance < 100) {
          const force = (100 - distance) / 100;
          ball.vx -= (dx / distance) * force * 0.5;
          ball.vy -= (dy / distance) * force * 0.5;
        }
      });
    };

    canvas.addEventListener("mousemove", handleMouseMove);

    return () => {
      window.removeEventListener("resize", resizeCanvas);
      canvas.removeEventListener("mousemove", handleMouseMove);
      if (animationRef.current) {
        cancelAnimationFrame(animationRef.current);
      }
    };
  }, []);

  return (
    <div className="flex flex-col items-center justify-center min-h-screen bg-gradient-to-br from-indigo-900 via-purple-900 to-pink-900 text-white overflow-hidden">
      <div className="text-center space-y-6 z-10 relative">
        <h2 className="text-3xl font-bold bg-gradient-to-r from-pink-400 to-violet-400 bg-clip-text text-transparent">
          Loading Portfolio
        </h2>
        <p className="text-purple-200 max-w-md">
          Move your mouse around to interact with the bouncing balls while we
          prepare your content
        </p>

        {/* Animated loading text */}
        <div className="flex items-center justify-center space-x-2">
          <span className="text-lg">Loading</span>
          <div className="flex space-x-1">
            {[0, 1, 2].map((i) => (
              <div
                key={i}
                className="w-2 h-2 bg-pink-400 rounded-full animate-bounce"
                style={{
                  animationDelay: `${i * 0.1}s`,
                  animationDuration: "0.6s",
                }}
              ></div>
            ))}
          </div>
        </div>
      </div>

      {/* Canvas for bouncing balls */}
      <canvas
        ref={canvasRef}
        className="absolute inset-0 w-full h-full cursor-pointer"
        style={{ width: "100%", height: "100%" }}
      />

      {/* Decorative elements */}
      <div className="absolute top-10 left-10 w-20 h-20 bg-gradient-to-br from-pink-500/20 to-violet-500/20 rounded-full blur-xl animate-pulse"></div>
      <div
        className="absolute bottom-10 right-10 w-32 h-32 bg-gradient-to-br from-violet-500/20 to-indigo-500/20 rounded-full blur-xl animate-pulse"
        style={{ animationDelay: "1s" }}

          <div className="flex justify-center space-x-1">
            {[0, 1, 2, 3].map((i) => (
              <div
                key={i}
                className="w-1.5 h-1.5 bg-indigo-400 rounded-full animate-bounce"
                style={{
                  animationDelay: `${i * 0.15}s`,
                  animationDuration: "1s",
                }}
              ></div>
            ))}
          </div>
        </div>

        {/* Completion State */}
        {loadingStage >= loadingMessages.length - 1 && (
          <div className="mt-6 p-4 bg-gradient-to-r from-green-50 to-emerald-50 border border-green-200 rounded-xl animate-fade-in">
            <div className="flex items-center justify-center space-x-2">
              <div className="w-5 h-5 bg-green-500 rounded-full flex items-center justify-center">
                <span className="text-white text-xs">✓</span>
              </div>
              <p className="text-green-700 font-medium">Portfolio Ready!</p>
            </div>
          </div>
        )}
      </div>

      {/* Custom CSS for fade-in animation */}
      <style jsx>{`
        @keyframes fade-in {
          from {
            opacity: 0;
            transform: translateY(10px);
          }
          to {
            opacity: 1;
            transform: translateY(0);
          }
        }
        .animate-fade-in {
          animation: fade-in 0.5s ease-out;
        }
      `}</style>
    </div>
  );
};

export default BrandedLogoAnimationLoader;
```

## Implementation Tips

### Choosing the Right Loading Component

- **Simple spinners**: Best for quick operations (< 3 seconds)
- **Progress bars**: Ideal for operations with measurable progress
- **Skeleton loaders**: Perfect for content that has a predictable layout
- **Branded loaders**: Great for initial app loading or page transitions

### Performance Considerations

```tsx
// Lazy load the loading component to reduce bundle size
const LazyLoading = React.lazy(() => import("@/components/Loading"));

// Use with Suspense for better UX
function App() {
  return (
    <Suspense fallback={<div>Loading...</div>}>
      <LazyLoading />
    </Suspense>
  );
}
```

### Using Loading States in Context

import React, { createContext, useContext, useState } from "react";

const LoadingContext = createContext();
